Aligning Spatial Transcriptomic DatasetsÔÉÅ

There are several packages currently available for aligning spatial transcriptomic datasets, including:PASTE, STalign, and SpatialData. Since our datasets come from serial tissue sections, we aligned the two datasets through manual landmark selection using Xenium Explorer. After registration, you can save the landmark points and transformation matrix.

H = np.array([[0.09405798593520641,3.41158653187994,-959.2731416178372],
             [3.41158653187994,-0.09405798593520641,-1528.8859782583281],
             [0,0,1]])

def align(H, adata_xe, adata_vis):

    ### Convert Visium spot coordinates to Xenium DAPI coordinates and scale
    points_vis = np.array(adata_vis.obs[["X_coords","Y_coords"]])
    points_vis2 = np.hstack((points_vis, np.ones((points_vis.shape[0], 1))))
    points_xe = np.dot(points_vis2, H.T)
    points_xe2 = points_xe[:, :2] / points_xe[:, 2, np.newaxis]
    points_xe2 = points_xe2*0.2125

    ### Check the alignment
    fig, axs = plt.subplots(1,2,dpi=100)
    axs[0].scatter(points_xe2[:,0], points_xe2[:,1],  marker='o', linewidth=0, alpha=1, color="black", s=15)
    axs[0].axis('scaled')
    axs[0].set_title("Visium")
    axs[1].scatter(adata_xe.obs["x_centroid"], adata_xe.obs["y_centroid"],  marker='o', linewidth=0, alpha=1, color="black", s=1)
    axs[1].axis('scaled')
    axs[1].set_title("Xenium")
    axs[0].invert_yaxis()
    axs[1].invert_yaxis()
    plt.show()

    return points_xe2

aligned_points = align(H, adata_xe, adata_vis)

adata_vis.obsm["spatial"] = aligned_points
